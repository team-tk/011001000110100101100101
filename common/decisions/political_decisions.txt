fascism_on_the_rise = {
	prepare_for_fascist_civil_war = {

		icon = generic_prepare_civil_war

		available = {
			fascism > 0.1
		}

		visible = {
			NOT = { has_country_flag = preparation_for_fascist_civil_war }
			has_idea_with_trait = fascist_demagogue
			check_has_focus_tree_to_switch_to_fascism = yes
			NOT = { has_fascist_government = yes }
		}
		
		fire_only_once = yes
		
		cost = 50
		
		complete_effect = {
			set_country_flag = preparation_for_fascist_civil_war
		}
		
		ai_will_do = {
			factor = 0
		}
	}
	
	expand_civil_fascism_support = {

		icon = generic_civil_support

		visible = {
			OR = {
				has_country_flag = preparation_for_fascist_civil_war
				has_country_flag = open_up_political_discourse_fascism
			}
			has_idea_with_trait = fascist_demagogue
			check_has_focus_tree_to_switch_to_fascism = yes
			NOT = { has_fascist_government = yes }
		}

		cancel_trigger = {
			hidden_trigger = {
				has_civil_war = yes
			}
		}
		
		fire_only_once = no
		
		cost = 25

		days_remove = 30

		modifier = {
			fascism_drift = 0.02
			stability_weekly = -0.03
 			political_power_gain = -0.2
		}
		
		ai_will_do = {
			factor = 1
		}
	}
	
	army_support_for_fascist_civil_war = {

		icon = generic_army_support

		visible = {
			has_country_flag = preparation_for_fascist_civil_war
			has_idea_with_trait = fascist_demagogue
			check_has_focus_tree_to_switch_to_fascism = yes
			NOT = { has_fascist_government = yes }
		}

		cancel_trigger = {
			hidden_trigger = {
				has_civil_war = yes
			}
		}
	
		fire_only_once = no
	
		cost = 50

		days_remove = 30

		modifier = {
			command_power_gain = -0.2
		}

		remove_effect = {
			if = {
				limit = {
					check_variable = { var = army_support_var value = 0.1 compare = less_than }
				}
				set_variable = { var = army_support_var value = 0.1 } 
			}
			if = {
				limit = {
					check_variable = { var = army_support_var value = 0.1 compare = greater_than_or_equals }
				}
				add_to_variable = { var = army_support_var value = 0.1 } 
			}
			set_country_flag = military_support_for_fascist_civil_war
			custom_effect_tooltip = army_support_TT
		}
		
		complete_effect = {
		}
		
		ai_will_do = {
			factor = 0
		}
	}

	ensure_general_loyalty_for_fascist_civil_war = {

		icon = generic_army_support

		available = {
			has_country_flag = military_support_for_fascist_civil_war

	siphon_equipment_stockpiles_for_fascist_civil_war = {

		icon = generic_prepare_civil_war

		available = {
			has_country_flag = military_support_for_fascist_civil_war
			has_equipment = { infantry_equipment > 1000 }
			command_power > 2
		}

		visible = {
			has_country_flag = preparation_for_fascist_civil_war
			has_country_flag = military_support_for_fascist_civil_war
			has_idea_with_trait = fascist_demagogue
			check_has_focus_tree_to_switch_to_fascism = yes
			NOT = { has_fascist_government = yes }

		}
		
		ai_will_do = {
			factor = 0
		}
	}
	
	ignite_the_fascist_civil_war = {

		icon = generic_ignite_civil_war

		available = {
			has_stability < 0.5
		}

		visible = {
			has_country_flag = preparation_for_fascist_civil_war
			has_idea_with_trait = fascist_demagogue
			check_has_focus_tree_to_switch_to_fascism = yes
			NOT = { has_fascist_government = yes }
		}
	
		fire_only_once = yes
	
		cost = 50
		
		complete_effect = {
			if = {
				limit = {
					has_democratic_government = yes
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@democratic
				}
			}
			if = {
				limit = {
					has_communist_government = yes
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@communism
				}
			}
			if = {
				limit = {
					has_neutral_government = yes
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@neutrality
				}
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = army_support_var
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ruling_party = fascism
				ideology = ROOT
				size = civil_war_size_var
				keep_unit_leaders_trigger = {
					has_trait = hidden_sympathies
				}
			}
			clr_country_flag = preparation_for_fascism_civil_war
			clr_country_flag = military_support_for_fascism_civil_war
			clr_country_flag = civil_support_for_fascism_civil_war
			set_country_flag = ideology_civil_war
		}
		
		ai_will_do = {
			factor = 0
		}
	}

	rebuild_the_nation_fascism = {

		icon = generic_construction

		visible = {
			has_country_flag = won_ideology_civil_war
		}

		fire_only_once = yes
	
		cost = 25

		days_remove = 365

		modifier = {
			political_power_gain = -1
			industry_repair_factor = 0.20
			production_speed_industrial_complex_factor = 0.1
			production_speed_infrastructure_factor = 0.1
			industrial_capacity_factory = 0.1
			industrial_capacity_dockyard = 0.1
		}
		
		ai_will_do = {
			factor = 200
		}
	}

	open_up_political_discourse_fascism = {

		icon = generic_political_discourse

		available = {
			fascism < 0.9
			democratic < 0.9
			communism < 0.9
			neutrality < 0.9
		}

		visible = {
			has_idea_with_trait = fascist_demagogue
			NOT = { has_country_flag = open_up_political_discourse_fascism }
			check_has_focus_tree_to_switch_to_fascism = yes
			NOT = { has_fascist_government = yes }
		}
		
		remove_trigger = {
			has_fascist_government = yes
		}
		
		fire_only_once = yes
		
		cost = 75
		
		complete_effect = {
			set_country_flag = open_up_political_discourse_fascism
		}
		
		ai_will_do = {
			factor = 1
		}
	}
	
	discredit_government_fascism = {

		icon = generic_political_discourse

		available = {
			fascism < 0.8
			democratic < 0.8
			communism < 0.8
			neutrality < 0.8
			has_idea_with_trait = fascist_demagogue
		}

		visible = {
			has_country_flag = open_up_political_discourse_fascism
			check_has_focus_tree_to_switch_to_fascism = yes
			NOT = { has_fascist_government = yes }
		}
		
		remove_trigger = {
			has_fascist_government = yes
		}
		
		fire_only_once = yes
		
		cost = 50
		
		complete_effect = {
			add_stability = -0.010
			if = {
				limit = {
					has_democratic_government = yes
				}
				add_popularity = {
					ideology = democratic
					popularity = -0.1
				}
			}
			if = {
				limit = {
					has_communist_government = yes
				}
				add_popularity = {
					ideology = communism
					popularity = -0.1
				}
			}
			if = {
				limit = {
					has_neutral_government = yes
				}
				add_popularity = {
					ideology = neutrality
					popularity = -0.1
				}
			}
		}
		
		ai_will_do = {
			factor = 1
		}
	}
	
	hold_the_fascist_national_referendum = {

		icon = generic_political_discourse

		available = {
			fascism > 0.5
		}

		visible = {
			NOT = { has_fascist_government = yes }
			check_has_focus_tree_to_switch_to_fascism = yes
			OR = {
				fascism > 0.6
				has_country_flag = open_up_political_discourse_fascism
			}
		}
		
		remove_trigger = {
			has_fascist_government = yes
		}
		
		fire_only_once = yes
		
		cost = 100
		
		complete_effect = {
			country_event = { id = political.18 }
		}
		
		ai_will_do = {
			factor = 210
			modifier = {
				fascism > 0.6
				factor = 10
			}
		}
	}
}
