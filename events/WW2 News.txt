# Rome Bombed
news_event = {
	id = ww2news.4
	title = ww2news.4.t
	desc = ww2news.4.d
	picture = GFX_news_event_german_reichstag_ruins

	fire_only_once = yes

	major = yes
	
	trigger = {
		tag = ITA
		date > 1943.1.1
		ITA = { is_in_faction_with = GER }
		ITA = { has_war_with = ENG }
		ITA = { has_war_with = SOV }
		ENG = { has_capitulated = no }
		SOV = { has_capitulated = no }
		
		OR = {
					ENG = { controls_state = 448 }
					any_other_country = {
						is_in_faction_with = ENG
						controls_state = 448
				}
				}
				
				OR = {
					ENG = { controls_state = 271 }
					any_other_country = {
						is_in_faction_with = ENG
						controls_state = 271
				}
				}
		
	}

	mean_time_to_happen = { 
		days = 1
	}
	
	immediate = {
		set_global_flag = rome_bombed
	}

	option = {
		name = ww2news.4.a	
	}
}
